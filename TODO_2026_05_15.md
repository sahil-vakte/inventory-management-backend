```python
def get_completion_percentage(self, obj):
    items = obj.items.all()
    total = items.count()
    if total == 0:
        return 0
    completed = items.filter(
        processing_status__in=['PICKED', 'COMPLETED']
```

### Endpoint

### 3.1 — Add custom action to `OrderItemViewSet` in `orders/views.py`

```python
from rest_framework.decorators import action
from django.contrib.auth.models import User

@action(detail=True, methods=['patch'], url_path='assign')
def assign(self, request, pk=None):
    item = self.get_object()
    employee_id = request.data.get('assigned_to')

    if employee_id is None:
        # Allow unassigning by passing null
        item.assigned_to = None
        item.save(update_fields=['assigned_to', 'updated_at'])
        return Response({
            'id': item.id,
            'sku': item.sku,
            'assigned_to': None,
            'assigned_to_username': None,
            'message': 'Item unassigned successfully'
        })

    try:
        employee = User.objects.get(pk=employee_id)
    except User.DoesNotExist:
        return Response(
            {'error': f'User with id {employee_id} does not exist'},
            status=status.HTTP_400_BAD_REQUEST
        )

    item.assigned_to = employee
    item.save(update_fields=['assigned_to', 'updated_at'])

    return Response({
        'id': item.id,
        'sku': item.sku,
        'order_number': item.order.order_number,
        'assigned_to': employee.id,
        'assigned_to_username': employee.username,
        'assigned_to_full_name': employee.get_full_name(),
        'message': f'Item assigned to {employee.username}'
    })
```

```python
@action(detail=True, methods=['patch'], url_path='update-status')
def update_status(self, request, pk=None):
    item = self.get_object()
    new_status = request.data.get('processing_status')
    quantity_processed = request.data.get('quantity_processed')

    valid_statuses = [s[0] for s in OrderItem.ITEM_STATUS_CHOICES]
    if new_status not in valid_statuses:
        return Response(
            {
                'error': f'Invalid status "{new_status}"',
                'valid_statuses': valid_statuses
            },
            status=status.HTTP_400_BAD_REQUEST
        )

    old_status = item.processing_status
    item.processing_status = new_status

    if quantity_processed is not None:
        if quantity_processed < 0 or quantity_processed > item.quantity:
            return Response(
                {'error': f'quantity_processed must be between 0 and {item.quantity}'},
                status=status.HTTP_400_BAD_REQUEST
            )
        item.quantity_processed = quantity_processed

    item.save(update_fields=['processing_status', 'quantity_processed', 'updated_at'])

    return Response({
        'id': item.id,
        'sku': item.sku,
        'order_number': item.order.order_number,
        'previous_status': old_status,
        'processing_status': item.processing_status,
        'processing_status_display': item.get_processing_status_display(),
        'quantity': item.quantity,
        'quantity_processed': item.quantity_processed,
        'completion_pct': round((item.quantity_processed / item.quantity) * 100) if item.quantity else 0
    })
```

## Notes

- **`assigned_to` on `Order` vs `OrderItem`:** The existing `Order.assigned_to` assigns the whole order to one employee. The new `OrderItem.assigned_to` assigns individual line items, allowing split-team picking (e.g. different pickers per SKU location).
- **`completion_percentage` logic:** Counts items by `processing_status` being `PICKED` or `COMPLETED` — not by `quantity_processed`. This is intentional because a picker marks an item as PICKED only when fully done. `quantity_processed` tracks incremental progress within an item.
- **`ITEM_STATUS_CHOICES` not in `WIMS_IMPLEMENTATION_TODO.md`:** This is implementing Phase 3.3 of the WIMS TODO early (it was planned as a future migration). Update the WIMS TODO to mark Phase 3.3 `OrderItem` fields as done after completing today's work.
- **No URL changes needed:** DRF `@action` decorators automatically register the new endpoints on the existing router — no edits to `orders/urls.py` required.
